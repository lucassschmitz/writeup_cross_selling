\begin{frame}{The Problem}
\begin{itemize}

\item Banks maximize the value of the customer, not of a particular product
\item Banks make losses on some products (Qi, 2024)
\item Existing customers are more likely to buy another product from the same bank (\cite{basten_cross-selling_2023})
\begin{itemize}
  \item Switching costs (\cite{klemperer_competition_1995,dube_switching_2009}) 
  \item Asymmetric information (\cite{sharpe_asymmetric_1990,petersen_effect_1995})
  \item Unobserved persistent heterogeneity
\end{itemize}
\item Understanding market power requires taking into account multi-product banks
\end{itemize}
\end{frame}


\begin{frame}{Research Idea}
\begin{itemize}
  \item Dynamic model of multi-product banks including: 
  \begin{itemize}
      \item Switching costs
      \item Asymmetric information: banks learn about their customers 
  \end{itemize}
\end{itemize}
\end{frame}
